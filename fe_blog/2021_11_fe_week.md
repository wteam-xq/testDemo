function findUniqueNumber(nums) {
    let memo = {};
    for (let i = 0; i < nums.length; i++) {
        const num = nums[i];
        if (!memo[num]) {
            memo[num] = 1;
        } else {
            memo[num] += 1;
        }
    }
    return Object.keys(memo).reduce((unique,num) => {
        return memo[num] === 1 ? Number(num) : unique;
    }, NaN);
};
